# ----------------------------------------------

#    sudo apt-get update && sudo apt-get upgrade
#
# 2. Install system dependencies:
#    sudo apt-get install -y python3-pip python3-dev
#    sudo apt-get install -y libatlas-base-dev libhdf5-dev
#    sudo apt-get install -y python3-opencv  # Pre-built OpenCV (faster)
#    sudo apt-get install -y espeak  # For text-to-speech
#    sudo apt-get install -y portaudio19-dev  # For audio
#    sudo apt-get install -y libsdl2-dev libsdl2-mixer-dev  # For pygame
#
# 3. Install Python packages:
#    pip3 install -r requirements.txt
#

#    pip3 install --index-url https://google-coral.github.io/py-repo/ tflite_runtime
